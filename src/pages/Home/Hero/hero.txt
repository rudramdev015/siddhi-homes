import React, { Suspense, memo } from "react";
import { Canvas } from "@react-three/fiber";
import { OrbitControls, useGLTF, Bounds } from "@react-three/drei";
import './Hero.css'; // Your hero section CSS

// --- STABLE MACBOOK COMPONENT ---
const MacBookModel = memo(() => {
    const { scene } = useGLTF("/macbook.glb");
    return <primitive object={scene} />;
});

// --- STABLE 3D SCENE COMPONENT ---
const ThreeScene = memo(() => {
    return (
        // === THIS IS THE FINAL UPDATE ===
        // The camera's y-position is now 0 for a perfectly straight-on view.
        // The z-position is adjusted slightly to frame it perfectly.
        <Canvas camera={{ position: [0, 0, 45], fov: 45 }}>
            <ambientLight intensity={0.5} />
            <directionalLight position={[5, 10, 5]} intensity={1.2} />
            <pointLight position={[-5, -5, -5]} intensity={0.3} />
            
            <Suspense fallback={null}>
                <Bounds fit clip margin={1.2}>
                    <MacBookModel />
                </Bounds>
            </Suspense>

            {/* Controls allow full freedom of rotation by the user */}
            <OrbitControls
                makeDefault
                enableZoom={false}
                enablePan={false}
            />
        </Canvas>
    );
});

// --- Your Main Hero Section Component ---
const HeroSection = () => {
    return (
        <section className="hero-section">
            <div className="hero-container">
                <div className="hero-left">
                    <ThreeScene />
                </div>
                <div className="hero-right">
                    <h2 className="hero-greeting">Hii, I am</h2>
                    <h1 className="hero-name">Jitendra Parmar</h1>
                    <p className="hero-description">
                        I create stunning, high-performance websites and have a passion for 3D visuals and modern web technologies.
                    </p>
                    <a href="/contact" className="hero-button">Get In Touch</a>
                </div>
            </div>
        </section>
    );
};

export default HeroSection;
